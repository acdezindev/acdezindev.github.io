Script Eagle - Enemigo con Patrulla (Waypoints)

¿Qué hace este script?
  • Controla el movimiento autónomo de un enemigo que patrulla entre varios puntos (waypoints)
  • El enemigo se mueve de un punto a otro de forma continua
  • Al llegar a un waypoint, cambia de dirección visual (flipX)
  • Implementa un bucle de patrulla que se repite indefinidamente
using UnityEngine;

public class Eagle : MonoBehaviour
{
    // Puntos de ruta (waypoints) para el movimiento
    [SerializeField] private Transform[] waypoint;
    [SerializeField] private float speed = 1f;

    // Índice del waypoint actual
    private int currentWaypointIndex = 0;

    // Para voltear el sprite
    private SpriteRenderer spriteRenderer;
    private int contadorFlips = 0;

    void Start()
    {
        spriteRenderer = GetComponent<SpriteRenderer>();
        // El sprite mira a izquierda por defecto (flipX = false)
        spriteRenderer.flipX = false;
    }

    void Update()
    {
        // Movimiento por transformación (no usa físicas)
        if (waypoint.Length > 0)
        {
            Transform currentWaypointTransform = waypoint[currentWaypointIndex];
            transform.position = Vector2.MoveTowards(transform.position, currentWaypointTransform.position, speed);
        }
    }

    private void OnTriggerEnter2D(Collider2D collision)
    {
        // Al llegar a un waypoint, avanza al siguiente
        if (collision.gameObject.CompareTag("waypoint"))
        {
            currentWaypointIndex = (currentWaypointIndex + 1) % waypoint.Length;
            contadorFlips++;

            // Control de giro según el waypoint alcanzado
            // false = mirando IZQUIERDA, true = mirando DERECHA
            if (contadorFlips == 1)
            {
                spriteRenderer.flipX = false;
            }
            else if (contadorFlips == 2)
            {
                spriteRenderer.flipX = true;
            }
            else if (contadorFlips == 3)
            {
                spriteRenderer.flipX = true;
            }
            else if (contadorFlips == 4)
            {
                spriteRenderer.flipX = false;
                contadorFlips = 0;
            }
        }
    }
}
