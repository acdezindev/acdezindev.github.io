Script GameManager - Gestor del Juego

¿Qué hace este script?
  • Gestiona el menú principal y la pantalla de ajustes
  • Controla la navegación entre escenas (menú → nivel)
  • Implementa el patrón Singleton para mantener una instancia global
  • Persiste entre escenas con DontDestroyOnLoad
  • Guarda las preferencias de música y efectos con PlayerPrefs
  • Gestiona el cierre de la aplicación (Editor y Build)
using UnityEngine;
using UnityEngine.SceneManagement;
using UnityEngine.UI;

public class GameManager : MonoBehaviour
{
    // Referencias a los Canvas de menú y ajustes
    [SerializeField] private GameObject menuJuego;
    [SerializeField] private GameObject ajustesJuego;

    // Instancia única (Singleton) y persistente entre escenas
    public static GameManager instance;

    // Variables globales para guardar estado del juego
    public int vidas;
    public int coleccionables;

    private void Awake()
    {
        // Patrón Singleton con DontDestroyOnLoad
        if (instance == null)
        {
            instance = this;
            DontDestroyOnLoad(gameObject);
        }
        else
        {
            Destroy(gameObject);
        }
    }

    void Start()
    {
        // Estado inicial: menú visible, ajustes ocultos
        menuJuego.SetActive(true);
        ajustesJuego.SetActive(false);
    }

    void Update()
    {
        // Vacío
    }

    // Carga el nivel principal
    public void cargarNivel1()
    {
        SceneManager.LoadScene("Nivel1");
    }

    // Cierra la aplicación
    public void SalirDelJuego()
    {
        if (Application.isEditor)
        {
            #if UNITY_EDITOR
            UnityEditor.EditorApplication.isPlaying = false;
            #endif
        }
        else
        {
            Application.Quit();
        }
    }

    // Control de navegación entre menú y ajustes
    public void VolverMenuAjustes()
    {
        menuJuego.SetActive(true);
        ajustesJuego.SetActive(false);
    }

    public void AbrirAjustes()
    {
        menuJuego.SetActive(false);
        ajustesJuego.SetActive(true);
    }

    public void CerrarAjustesVolver()
    {
        menuJuego.SetActive(true);
        ajustesJuego.SetActive(false);
    }

    // Guardado de preferencias de música
    public void MusicaON()
    {
        PlayerPrefs.SetInt("MusicaActiva", 1);
        PlayerPrefs.Save();
    }

    public void MusicaOFF()
    {
        PlayerPrefs.SetInt("MusicaActiva", 0);
        PlayerPrefs.Save();
    }

    // Guardado de preferencias de efectos
    public void EfectosON()
    {
        PlayerPrefs.SetInt("EfectosActivos", 1);
        PlayerPrefs.Save();
    }

    public void EfectosOFF()
    {
        PlayerPrefs.SetInt("EfectosActivos", 0);
        PlayerPrefs.Save();
    }
}
