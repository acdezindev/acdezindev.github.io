Script Zodito - Controlador del Jugador

¿Qué hace este script?
  • Controla el movimiento del personaje mediante entrada táctil (móviles)
  • Gestiona el salto y la detección del suelo
  • Controla las animaciones (reposo, carrera)
  • Gestiona la recogida de gemas y el contador
  • Controla el sistema de vidas (3 vidas iniciales)
  • Maneja colisiones con enemigos (pérdida de vida)
  • Gestiona la victoria (10 gemas) y derrota (0 vidas)
  • Reproduce efectos de sonido con PlayerPrefs
using System;
using UnityEngine;
using UnityEngine.Scripting.APIUpdating;
using TMPro;
using UnityEngine.SceneManagement;

public class Zodito : MonoBehaviour
{
    // Velocidad de movimiento del personaje
    [SerializeField] private float speed = 3f;

    // Fuerza aplicada al saltar
    [SerializeField] private float salto = 3f;

    // Referencia al texto UI para mostrar las gemas recogidas
    [SerializeField] private TextMeshProUGUI contadorGemasText;

    // Referencia al texto UI para mostrar las vidas restantes
    [SerializeField] private TextMeshProUGUI contadorvidasText;

    // Referencia al texto UI para mostrar mensajes temporales
    [SerializeField] private TextMeshProUGUI textoMensaje;

    // Componentes necesarios para el control del personaje
    private Rigidbody2D rb;
    private SpriteRenderer spriteRenderer;
    private Animator animator;

    // Contador de gemas recolectadas
    private int contadorGemas = 0;

    // Vidas del jugador, valor inicial de 3 vidas
    private int contadorVidas = 3;

    // Control para permitir salto solo cuando está en el suelo
    private bool isSuelo = false;

    // Fuente de audio para efectos de sonido
    private AudioSource audioSource;

    // Clips de audio para diferentes eventos del juego
    public AudioClip sonidoEliminar;
    public AudioClip sonidoGemas;
    public AudioClip sonidofinalPantalla;
    public AudioClip sonidoPerderTodasLasVidas;
    public AudioClip sonidoPerderVida;

    void Start()
    {
        // Inicialización de componentes
        rb = GetComponent<Rigidbody2D>();
        spriteRenderer = GetComponent<SpriteRenderer>();
        animator = GetComponent<Animator>();

        // Actualización inicial de la UI con los valores actuales
        contadorGemasText.text = contadorGemas.ToString();
        contadorvidasText.text = contadorVidas.ToString();

        // Configuración del sistema de audio
        audioSource = GetComponent<AudioSource>();
    }

    void Update()
    {
        // Control de animación: activa "isRunning" si hay movimiento horizontal
        animator.SetBool("isRunning", Math.Abs(rb.linearVelocityX) > 0.2);

        // Detección de entrada táctil para móviles
        // Más de un toque = salto
        if (Input.touchCount > 1) saltamos();

        // Un toque = movimiento
        if (Input.touchCount > 0)
        {
            UnityEngine.Touch touch = Input.GetTouch(0);
            if (touch.phase == TouchPhase.Stationary)
            {
                Move(touch);
            }
        }
    }

    // Gestión de colisiones con triggers (coleccionables y zonas especiales)
    void OnTriggerEnter2D(Collider2D collision)
    {
        // Recolección de gemas
        if (collision.gameObject.CompareTag("diamante"))
        {
            EfectosActivosClip(sonidoGemas);
            Destroy(collision.gameObject);
            contadorGemas++;
            contadorGemasText.text = contadorGemas.ToString();
        }

        // Eliminación de enemigos al saltar sobre ellos
        if (collision.gameObject.CompareTag("eliminar"))
        {
            EfectosActivosClip(sonidoEliminar);
            Destroy(collision.gameObject.transform.parent.gameObject);
        }

        // Verificación de condición de victoria al llegar al final del nivel
        if (collision.gameObject.CompareTag("final"))
        {
            // Condición de victoria: 10 gemas recolectadas
            if (contadorGemas >= 10)
            {
                EfectosActivosClip(sonidofinalPantalla);
                
                // Guardado de datos persistentes entre escenas
                PlayerPrefs.SetInt("VidasFinales", contadorVidas);
                PlayerPrefs.SetInt("ColeccionablesFinales", contadorGemas);
                PlayerPrefs.Save();
                
                SceneManager.LoadScene("Win");
            }
            else
            {
                // Mensaje informativo si no se cumplen las condiciones
                if (textoMensaje != null)
                {
                    int gemasFaltan = 10 - contadorGemas;
                    textoMensaje.text = "[!] GEMAS INSUFICIENTES [!]" + "\n"
                    + "Te faltan " + gemasFaltan + " gemas para avanzar de nivel" + "\n"
                    + "Llevas " + contadorGemas + " gemas" + "\n"
                    + "¡Vuelve atras y busca mas!!!! !";
                    textoMensaje.gameObject.SetActive(true);
                    Invoke("OcultarMensaje", 4f);
                }
            }
        }
    }

    // Oculta el mensaje de advertencia después de un tiempo
    private void OcultarMensaje()
    {
        if (textoMensaje != null)
        {
            textoMensaje.gameObject.SetActive(false);
        }
    }

    // Control de movimiento basado en entrada táctil
    private void Move(UnityEngine.Touch t)
    {
        // División de pantalla en zonas: izquierda/derecha
        if (t.position.x > Screen.width / 3)
        {
            // Movimiento a la derecha
            rb.linearVelocity = new Vector2(speed, rb.linearVelocityY);
            spriteRenderer.flipX = false;
        }
        else
        {
            // Movimiento a la izquierda
            rb.linearVelocity = new Vector2(-speed, rb.linearVelocityY);
            spriteRenderer.flipX = true;
        }
    }

    // Ejecuta el salto del personaje
    private void saltamos()
    {
        // Solo permite saltar si está en el suelo
        if (isSuelo == true)
        {
            rb.linearVelocity = new Vector2(rb.linearVelocity.x, salto);

            // Reproducción de sonido condicionada a preferencias de usuario
            if (PlayerPrefs.GetInt("EfectosActivos", 1) == 1)
            {
                audioSource.Play();
            }
        }
    }

    // Detección de colisiones físicas con el entorno y enemigos
    private void OnCollisionEnter2D(Collision2D collision)
    {
        // Detección de contacto con el suelo para permitir salto
        if (collision.gameObject.CompareTag("suelo"))
        {
            isSuelo = true;
        }

        // Gestión de daño por colisión con enemigos
        if (collision.gameObject.CompareTag("enemigo"))
        {
            // Pequeño rebote al recibir daño
            rb.linearVelocity = new Vector2(rb.linearVelocity.x, 3f);

            if (contadorVidas > 0)
            {
                contadorVidas--;
                EfectosActivosClip(sonidoPerderVida);
                contadorvidasText.text = contadorVidas.ToString();

                // Condición de derrota: sin vidas restantes
                if (contadorVidas <= 0)
                {
                    EfectosActivosClip(sonidoPerderTodasLasVidas);
                    
                    // Guardado de datos antes de cambiar de escena
                    PlayerPrefs.SetInt("VidasFinales", contadorVidas);
                    PlayerPrefs.SetInt("ColeccionablesFinales", contadorGemas);
                    PlayerPrefs.Save();

                    SceneManager.LoadScene("GameOver");
                }
            }
        }
    }

    // Detección de fin de contacto con el suelo
    private void OnCollisionExit2D(Collision2D collision)
    {
        if (collision.gameObject.CompareTag("suelo"))
        {
            isSuelo = false;
        }
    }

    // Reproduce efecto de sonido directamente desde el AudioSource
    private void EfectosActivos()
    {
        if (PlayerPrefs.GetInt("EfectosActivos", 1) == 1)
        {
            audioSource.Play();
        }
    }

    // Reproduce un clip de audio específico usando PlayOneShot
    // Útil para sonidos que se superponen sin interrumpir otros
    private void EfectosActivosClip(AudioClip sonido)
    {
        if (PlayerPrefs.GetInt("EfectosActivos", 1) == 1)
        {
            audioSource.PlayOneShot(sonido);
        }
    }
}
